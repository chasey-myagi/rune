# Rune v0.3.0: Data Capabilities Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox syntax for tracking.

**Goal:** Add data capabilities to Rune Runtime — schema validation, file transfer, and the protocol extensions to support them.

**Architecture:** Three modules layered in dependency order: (1) Module 11 extends the protobuf protocol with `input_schema`/`output_schema` fields, `FileAttachment` message, and `priority` field; (2) Module 4 adds a `rune-schema` crate using `jsonschema` for JSON Schema validation at the Gate layer, plus OpenAPI auto-generation; (3) Module 3 adds file transfer via multipart upload, with small files inlined in gRPC attachments and large files brokered through a temporary file store.

**Tech Stack:** Rust, prost/tonic (proto), jsonschema (validation), serde_json (OpenAPI), axum-multipart/multer (file upload), tempfile (temp storage), uuid (file IDs)

**Spec:** `docs/superpowers/specs/2026-03-22-rune-v1.0.0-design.md` — Modules 3, 4, 11

```rust
use jsonschema::Validator;
use serde_json::Value;

#[derive(Debug, thiserror::Error)]
pub enum SchemaError {
    #[error("invalid JSON input: {0}")]
    InvalidJson(String),

    #[error("invalid JSON Schema: {0}")]
    InvalidSchema(String),

    #[error("validation failed: {errors:?}")]
    ValidationFailed { errors: Vec<String> },
}

/// Validate input bytes against a JSON Schema string.
/// Empty schema string means "no validation" (always passes).
pub fn validate_input(schema: &str, input: &[u8]) -> Result<(), SchemaError> {
    validate(schema, input)
}

/// Validate output bytes against a JSON Schema string.
/// Empty schema string means "no validation" (always passes).
pub fn validate_output(schema: &str, output: &[u8]) -> Result<(), SchemaError> {
    validate(schema, output)
}

fn validate(schema_str: &str, data: &[u8]) -> Result<(), SchemaError> {
    if schema_str.is_empty() {
        return Ok(());
    }

    let schema: Value = serde_json::from_str(schema_str)
        .map_err(|e| SchemaError::InvalidSchema(e.to_string()))?;

    let instance: Value = serde_json::from_slice(data)
        .map_err(|e| SchemaError::InvalidJson(e.to_string()))?;

    let validator = Validator::new(&schema)
        .map_err(|e| SchemaError::InvalidSchema(e.to_string()))?;

    let result = validator.validate(&instance);
    if let Err(errors) = result {
        let error_messages: Vec<String> = errors
            .map(|e| format!("{} at {}", e, e.instance_path))
            .collect();
        return Err(SchemaError::ValidationFailed { errors: error_messages });
    }

    Ok(())
}
```

```rust
use serde_json::{json, Value};

/// Information about a rune endpoint for OpenAPI generation.
pub struct RuneEndpointInfo {
    pub name: String,
    pub description: String,
    pub gate_path: Option<String>,
    pub gate_method: String,
    pub input_schema: Option<String>,
    pub output_schema: Option<String>,
}

/// Generate an OpenAPI 3.0.3 JSON document from registered rune endpoints.
pub fn generate_openapi(endpoints: &[RuneEndpointInfo], title: &str, version: &str) -> Value {
    let mut paths = serde_json::Map::new();

    for ep in endpoints {
        let path = ep.gate_path.clone()
            .unwrap_or_else(|| format!("/api/v1/runes/{}/run", ep.name));
        let method = ep.gate_method.to_lowercase();

        let mut request_body = json!({
            "required": true,
            "content": {
                "application/json": {
                    "schema": {}
                }
            }
        });

        if let Some(ref schema_str) = ep.input_schema {
            if let Ok(schema) = serde_json::from_str::<Value>(schema_str) {
                request_body["content"]["application/json"]["schema"] = schema;
            }
        }

        let mut success_response = json!({
            "description": "Successful response"
        });

        if let Some(ref schema_str) = ep.output_schema {
            if let Ok(schema) = serde_json::from_str::<Value>(schema_str) {
                success_response["content"] = json!({
                    "application/json": {
                        "schema": schema
                    }
                });
            }
        }

        let operation = json!({
            "summary": ep.name,
            "description": ep.description,
            "requestBody": request_body,
            "responses": {
                "200": success_response,
                "422": {
                    "description": "Validation failed"
                },
                "404": {
                    "description": "Rune not found"
                }
            }
        });

        let path_item = paths.entry(path).or_insert_with(|| json!({}));
        path_item[method] = operation;
    }

    json!({
        "openapi": "3.0.3",
        "info": {
            "title": title,
            "version": version,
        },
        "paths": Value::Object(paths),
    })
}
```

```rust
pub fn list_with_schemas(&self) -> Vec<(String, String, Option<String>, Option<String>, Option<String>, Option<String>)> {
    // Returns: (name, description, gate_path, gate_method, input_schema, output_schema)
    let mut result = Vec::new();
    for entry in self.entries.iter() {

                gate_path,
                gate_method,
                e.config.input_schema.clone(),
                e.config.output_schema.clone(),

    }
```

```rust
use std::collections::HashMap as StdHashMap;
use std::sync::RwLock;

/// In-memory file broker for temporary file storage.
pub struct FileBroker {
    files: RwLock<StdHashMap<String, StoredFile>>,
    max_size_bytes: usize,
}

struct StoredFile {
    data: Vec<u8>,
    filename: String,
    mime_type: String,
}

impl FileBroker {
    pub fn new(max_upload_size_mb: u64) -> Self {
        Self {
            files: RwLock::new(StdHashMap::new()),
            max_size_bytes: (max_upload_size_mb * 1024 * 1024) as usize,
        }
    }

    pub fn store(&self, filename: String, data: Vec<u8>, mime_type: String) -> Result<String, &'static str> {
        if data.len() > self.max_size_bytes {
            return Err("file too large");
        }
        let file_id = uuid::Uuid::new_v4().to_string();
        self.files.write().unwrap().insert(file_id.clone(), StoredFile { data, filename, mime_type });
        Ok(file_id)
    }

    pub fn get(&self, file_id: &str) -> Option<(String, Vec<u8>, String)> {
        let files = self.files.read().unwrap();
        files.get(file_id).map(|f| (f.filename.clone(), f.data.clone(), f.mime_type.clone()))
    }

    pub fn remove(&self, file_id: &str) -> bool {
        self.files.write().unwrap().remove(file_id).is_some()
    }
}
```

```rust
async fn upload_file(
    State(state): State<GateState>,
    req: axum::extract::Request,
) -> impl IntoResponse {
    let content_type = req.headers()
        .get("content-type")
        .and_then(|v| v.to_str().ok())
        .unwrap_or("")
        .to_string();

    let boundary = multer::parse_boundary(&content_type)
        .map_err(|_| error_response(StatusCode::BAD_REQUEST, "BAD_REQUEST", "missing multipart boundary"));

    let boundary = match boundary {
        Ok(b) => b,
        Err(resp) => return resp,
    };

    let body = req.into_body();
    let body_stream = body.into_data_stream();
    let mut multipart = multer::Multipart::new(body_stream, boundary);

    let max_bytes = state.max_upload_size_mb * 1024 * 1024;

    while let Ok(Some(field)) = multipart.next_field().await {
        let filename = field.file_name().unwrap_or("unnamed").to_string();
        let mime = field.content_type().unwrap_or("application/octet-stream").to_string();

        match field.bytes().await {
            Ok(data) => {
                if data.len() as u64 > max_bytes {
                    return error_response(StatusCode::PAYLOAD_TOO_LARGE, "PAYLOAD_TOO_LARGE", "file exceeds size limit");
                }
                match state.file_broker.store(filename, data.to_vec(), mime) {
                    Ok(file_id) => {
                        return (StatusCode::CREATED, Json(serde_json::json!({"file_id": file_id}))).into_response();
                    }
                    Err(e) => {
                        return error_response(StatusCode::PAYLOAD_TOO_LARGE, "PAYLOAD_TOO_LARGE", e);
                    }
                }
            }
            Err(_) => {
                return error_response(StatusCode::BAD_REQUEST, "BAD_REQUEST", "failed to read file data");
            }
        }
    }

    error_response(StatusCode::BAD_REQUEST, "BAD_REQUEST", "no file found in multipart body")
}

async fn download_file(
    State(state): State<GateState>,
    Path(id): Path<String>,
) -> impl IntoResponse {
    match state.file_broker.get(&id) {
        Some((filename, data, mime_type)) => {
            let headers = [
                ("content-type", mime_type),
                ("content-disposition", format!("attachment; filename=\"{}\"", filename)),
            ];
            (StatusCode::OK, headers, data).into_response()
        }
        None => error_response(StatusCode::NOT_FOUND, "NOT_FOUND", "file not found"),
    }
}
```
