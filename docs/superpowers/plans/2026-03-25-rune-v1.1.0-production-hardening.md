    let schema = r#"{"type": "object"}"#;
    assert!(rune_schema::validate_input(Some(schema), b"{}").is_ok());

    let result = std::panic::catch_unwind(|| {
        rune_schema::validate_input(Some(r#"{"type": "string"}"#), b"\"hello\"")

    assert!(result.is_ok());

- [ ] **Step 2: 修改 validator.rs — unwrap → poison recovery**

```rust
fn get_or_compile(schema_str: &str) -> Result<Arc<jsonschema::Validator>, SchemaError> {
    // Fast path: read lock
    {
        let cache = cache().read().unwrap_or_else(|e| e.into_inner());
        if let Some(v) = cache.get(schema_str) {
            return Ok(Arc::clone(v));
        }
    }

    // Slow path: compile and insert with write lock
    let schema_value: serde_json::Value = serde_json::from_str(schema_str)
        .map_err(|e| SchemaError::InvalidSchema(e.to_string()))?;
    let validator = jsonschema::validator_for(&schema_value)
        .map_err(|e| SchemaError::InvalidSchema(e.to_string()))?;
    let v = Arc::new(validator);

    let mut cache = cache().write().unwrap_or_else(|e| e.into_inner());
    cache.insert(schema_str.to_string(), Arc::clone(&v));
    Ok(v)
}

pub fn clear_validator_cache() {
    if let Some(c) = VALIDATOR_CACHE.get() {
        c.write().unwrap_or_else(|e| e.into_inner()).clear();
    }
}
```

### Task 3: Flow sync 模式释放 RwLock

**Files:**
- Modify: `runtime/rune-gate/src/handlers/flow.rs` (run_flow handler 中 sync 模式分支)

**问题:** `run_flow` 的 sync 模式先 clone 了 `flow_def`，但执行时仍通过 `engine.execute_flow()` 调用，而 `engine` 是从 `state.flow_engine.read().await` 获取的读锁引用。`execute_flow` 期间持有读锁，阻塞 `create_flow` 写操作。

`FlowEngine::execute_flow` 已有 `pub async fn execute_flow(&self, flow: &FlowDefinition, input: Bytes)` 签名，接受 pre-fetched `FlowDefinition`。关键问题是它需要 `&self` 来访问 `self.relay` 和 `self.resolver`。

- [ ] **Step 1: 检查 flow handler 中 sync 模式如何调用 execute_flow**

读取 flow handler 的 sync 分支代码，确认调用模式。

- [ ] **Step 2: 重构 — clone engine 所需的 Arc 字段，释放锁后再执行**

在 flow handler sync 模式中，改为：

```rust
// Clone flow_def AND engine's internal refs, then release the lock
let (flow_def, engine_clone) = {
    let engine = state.flow_engine.read().await;
    let flow = engine.get(&name).ok_or_else(|| /* ... */)?;
    (flow.clone(), engine.clone_runner())
};
// Execute without holding any lock
let result = engine_clone.execute_flow(&flow_def, body).await;
```

这需要在 `FlowEngine` 上添加一个 `clone_runner()` 方法，返回一个轻量的 `FlowRunner` struct（只包含 `Arc<Relay>` + `Arc<dyn Resolver>`）。

- [ ] **Step 3: 在 FlowEngine 中添加 FlowRunner**

在 `runtime/rune-flow/src/engine.rs` 中：

```rust
/// Lightweight handle for executing flows without holding engine lock.
pub struct FlowRunner {
    pub relay: Arc<Relay>,
    pub resolver: Arc<dyn Resolver>,
    pub step_timeout: Duration,  // 不要漏掉 — execute_flow 内部用它控制单步超时
}

impl FlowEngine {
    /// Create a lightweight runner that can execute flows without the engine lock.
    pub fn clone_runner(&self) -> FlowRunner {
        FlowRunner {
            relay: Arc::clone(&self.relay),
            resolver: Arc::clone(&self.resolver),
            step_timeout: self.step_timeout,
        }
    }
}
```

将 `execute_flow` 的核心逻辑移到 `FlowRunner` 上（或者让 `FlowEngine::execute_flow` 内部委托给 `FlowRunner`）。注意：`execute_flow` 不再访问 `self.flows`（它接受 `&FlowDefinition` 参数），只需要 `relay` + `resolver` + `step_timeout`。

let mut p95_stmt = conn.prepare(
    "WITH ranked AS (
        SELECT rune_name, latency_ms,
               ROW_NUMBER() OVER (PARTITION BY rune_name ORDER BY latency_ms ASC) as rn,
               COUNT(*) OVER (PARTITION BY rune_name) as cnt

type HmacSha256 = Hmac<Sha256>;

fn hmac_key(raw_key: &str, secret: &[u8]) -> String {
    let mut mac = HmacSha256::new_from_slice(secret)

### Task 7: Session 超时 — 用 per-request tokio::time::timeout 替代轮询

**Files:**
- Modify: `runtime/rune-core/src/session.rs:169-190`
- Test: existing session tests

**问题:** 每个 caster session 每秒 O(n) 扫描所有 pending requests。10 个 caster 各 1000 个请求 = 每秒 10000 次 DashMap 遍历。

- [ ] **Step 1: 分析当前超时机制的调用点**

确认 `timeout_handle` 只在 `handle_session` 中使用，且 `PendingRequest` 的 timeout 字段在创建时就确定。

- [ ] **Step 2: 改为在 dispatch_request 时注册 per-request timeout**

不再用全局轮询，改为在每个请求 dispatch 时 spawn 一个 `tokio::time::sleep(timeout)` future：

```rust
// 在 dispatch_request 之后，为每个 request spawn 一个 timeout task
let timeout_pending = Arc::clone(&pending);
let req_id_clone = request_id.clone();
let timeout_dur = request_timeout;
tokio::spawn(async move {
    tokio::time::sleep(timeout_dur).await;
    if let Some((_, p)) = timeout_pending.remove(&req_id_clone) {
        tracing::warn!(request_id = %req_id_clone, "request timed out");
        match p.tx {
            PendingResponse::Once(tx) => { let _ = tx.send(Err(RuneError::Timeout)); }
            PendingResponse::Stream(tx) => { let _ = tx.send(Err(RuneError::Timeout)).await; }
        }
    }
});
```

删除原来的 timeout scanner loop (第 170-190 行)。

注意：当请求正常完成时，`pending.remove(&req_id)` 已经发生，timeout task 的 `remove` 会返回 None，自然跳过。

### Task 10: FileBroker 大文件落磁盘

**Files:**
- Modify: `runtime/rune-gate/src/file_broker.rs`
- Modify: `runtime/rune-core/src/config.rs` (新增 `gate.file_broker_dir` 配置)

**问题:** FileBroker 全内存存储。多个并发大文件上传快速耗尽内存。

- [ ] **Step 1: 添加磁盘后端**

设计思路：小文件（< 4MB，已有的 `INLINE_THRESHOLD`）仍然走内存，大文件写入临时目录。

```rust
pub enum FileStorage {
    Memory(Bytes),
    Disk { path: PathBuf, size: u64 },
}

pub struct StoredFile {
    pub filename: String,
    pub mime_type: String,
    pub storage: FileStorage,
    pub request_id: String,
    pub stored_at: Instant,
}

impl StoredFile {
    pub fn data(&self) -> std::io::Result<Bytes> {
        match &self.storage {
            FileStorage::Memory(data) => Ok(data.clone()),
            FileStorage::Disk { path, .. } => {
                Ok(Bytes::from(std::fs::read(path)?))
            }
        }
    }
}
```

- [ ] **Step 2: store() 方法根据大小选择存储方式**

```rust
const DISK_THRESHOLD: usize = 4 * 1024 * 1024; // 4MB

pub fn store(&self, filename: String, mime_type: String, data: Bytes, request_id: &str) -> String {
    self.maybe_evict_expired();
    let file_id = uuid::Uuid::new_v4().to_string();
    let storage = if data.len() > DISK_THRESHOLD {
        let path = self.disk_dir.join(&file_id);
        std::fs::write(&path, &data).expect("failed to write file to disk");
        FileStorage::Disk { path, size: data.len() as u64 }
    } else {
        FileStorage::Memory(data)
    };
    // ...
}
```

- [ ] **Step 3: evict_expired 清理磁盘文件**

```rust
fn evict_expired(&self) {
    let ttl = self.ttl_secs;
    self.files.retain(|_, v| {
        if v.stored_at.elapsed().as_secs() >= ttl {
            if let FileStorage::Disk { ref path, .. } = v.storage {
                let _ = std::fs::remove_file(path);
            }
            false
        } else {
            true
        }
    });
}
```

```rust
// runtime/rune-core/src/telemetry.rs

use tracing_subscriber::{layer::SubscriberExt, util::SubscriberInitExt, EnvFilter};

pub struct TelemetryConfig {
    pub otlp_endpoint: Option<String>,
    pub prometheus_port: Option<u16>,
}

pub fn init_telemetry(config: &TelemetryConfig) {
    let env_filter = EnvFilter::try_from_default_env()
        .unwrap_or_else(|_| EnvFilter::new("info"));

    let fmt_layer = tracing_subscriber::fmt::layer();

    let registry = tracing_subscriber::registry()
        .with(env_filter)
        .with(fmt_layer);

    // 如果配置了 OTLP endpoint，添加 tracing exporter
    if let Some(ref endpoint) = config.otlp_endpoint {
        let tracer = opentelemetry_otlp::new_pipeline()
            .tracing()
            .with_exporter(
                opentelemetry_otlp::new_exporter()
                    .tonic()
                    .with_endpoint(endpoint),
            )
            .install_batch(opentelemetry_sdk::runtime::Tokio)
            .expect("failed to install OTLP tracer");
        let otel_layer = tracing_opentelemetry::layer().with_tracer(tracer);
        registry.with(otel_layer).init();
    } else {
        registry.init();
    }

    // 如果配置了 Prometheus port，启动 metrics exporter
    if let Some(port) = config.prometheus_port {
        let builder = metrics_exporter_prometheus::PrometheusBuilder::new();
        builder
            .with_http_listener(([0, 0, 0, 0], port))
            .install()
            .expect("failed to install Prometheus exporter");
    }
}
```
