# Rune v0.2.0: Infrastructure Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add configuration system, SQLite persistence, and API Key authentication to Rune Runtime, making it deployable on a cloud server.

**Architecture:** Three new capabilities layered on v0.1.0: (1) TOML config replaces hardcoded `AppConfig`, (2) `rune-store` crate wraps SQLite for tasks/keys/logs/snapshots, (3) auth middleware in Gate + gRPC validates API Keys. Config is built first since store and auth depend on its settings.

**Tech Stack:** Rust, toml + serde (config), rusqlite (SQLite), sha2 (key hashing), axum middleware (auth), clap (minimal CLI flags)

**Spec:** `docs/superpowers/specs/2026-03-22-rune-v1.0.0-design.md` — Modules 1, 2, 8

```rust
impl AppConfig {
    /// Load from TOML file. Missing fields use defaults.
    pub fn from_file(path: &str) -> anyhow::Result<Self> {
        let content = std::fs::read_to_string(path)?;
        let config: AppConfig = toml::from_str(&content)?;
        Ok(config)
    }

    /// Load with priority: CLI args > env vars > file > defaults.
    /// `file_path` is optional; if None, tries ~/.rune/config.toml.
    pub fn load(file_path: Option<&str>) -> Self {
        let mut config = match file_path {
            Some(path) => Self::from_file(path).unwrap_or_default(),
            None => {
                let default_path = dirs::home_dir()
                    .unwrap_or_default()
                    .join(".rune")
                    .join("config.toml");
                if default_path.exists() {
                    Self::from_file(default_path.to_str().unwrap()).unwrap_or_default()
                } else {
                    Self::default()
                }
            }
        };

        // Environment variable overrides (RUNE_*)
        if let Ok(v) = std::env::var("RUNE_GRPC_PORT") {
            if let Ok(p) = v.parse() { config.server.grpc_port = p; }
        }
        if let Ok(v) = std::env::var("RUNE_HTTP_PORT") {
            if let Ok(p) = v.parse() { config.server.http_port = p; }
        }
        if let Ok(v) = std::env::var("RUNE_DEV_MODE") {
            config.server.dev_mode = v == "true" || v == "1";
        }
        if let Ok(v) = std::env::var("RUNE_DB_PATH") {
            config.store.db_path = v;
        }
        if let Ok(v) = std::env::var("RUNE_LOG") {
            config.log.level = v;
        }
        if let Ok(v) = std::env::var("RUNE_AUTH_ENABLED") {
            config.auth.enabled = v == "true" || v == "1";
        }

        config
    }

    /// Apply --dev flag overrides.
    pub fn apply_dev_mode(&mut self) {
        self.server.dev_mode = true;
        self.server.grpc_host = "127.0.0.1".into();
        self.server.http_host = "127.0.0.1".into();
        self.auth.enabled = false;
    }

    /// Get gRPC SocketAddr.
    pub fn grpc_addr(&self) -> SocketAddr {
        format!("{}:{}", self.server.grpc_host, self.server.grpc_port)
            .parse()
            .expect("invalid gRPC address")
    }

    /// Get HTTP SocketAddr.
    pub fn http_addr(&self) -> SocketAddr {
        format!("{}:{}", self.server.http_host, self.server.http_port)
            .parse()
            .expect("invalid HTTP address")
    }

    /// Get heartbeat interval as Duration.
    pub fn heartbeat_interval(&self) -> Duration {
        Duration::from_secs(self.session.heartbeat_interval_secs)
    }

    /// Get heartbeat timeout as Duration.
    pub fn heartbeat_timeout(&self) -> Duration {
        Duration::from_secs(self.session.heartbeat_timeout_secs)
    }
}
```

Create `runtime/rune-store/src/store.rs`:
```rust
use rusqlite::{Connection, params};
use std::path::Path;
use std::sync::Mutex;
use tracing::info;

pub struct RuneStore {
    conn: Mutex<Connection>,
}

impl RuneStore {
    pub fn open(db_path: &str) -> Result<Self, rusqlite::Error> {
        // Ensure parent directory exists
        if let Some(parent) = Path::new(db_path).parent() {
            std::fs::create_dir_all(parent).ok();
        }

        let conn = Connection::open(db_path)?;
        conn.execute_batch("PRAGMA journal_mode=WAL;")?;
        conn.execute_batch("PRAGMA foreign_keys=ON;")?;

        let store = Self { conn: Mutex::new(conn) };
        store.migrate()?;
        Ok(store)
    }

    pub fn open_in_memory() -> Result<Self, rusqlite::Error> {
        let conn = Connection::open_in_memory()?;
        let store = Self { conn: Mutex::new(conn) };
        store.migrate()?;
        Ok(store)
    }

    fn migrate(&self) -> Result<(), rusqlite::Error> {
        let conn = self.conn.lock().unwrap();
        conn.execute_batch(
            "CREATE TABLE IF NOT EXISTS api_keys (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                key_prefix TEXT NOT NULL,
                key_hash TEXT NOT NULL UNIQUE,
                key_type TEXT NOT NULL CHECK(key_type IN ('caster', 'gate')),
                label TEXT NOT NULL DEFAULT '',
                created_at TEXT NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ', 'now')),
                revoked_at TEXT
            );

            CREATE TABLE IF NOT EXISTS tasks (
                task_id TEXT PRIMARY KEY,
                rune_name TEXT NOT NULL,
                status TEXT NOT NULL DEFAULT 'pending'
                    CHECK(status IN ('pending', 'running', 'completed', 'failed', 'cancelled')),
                input TEXT,
                output TEXT,
                error TEXT,
                created_at TEXT NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ', 'now')),
                started_at TEXT,
                completed_at TEXT
            );
            CREATE INDEX IF NOT EXISTS idx_tasks_status ON tasks(status);
            CREATE INDEX IF NOT EXISTS idx_tasks_rune ON tasks(rune_name);

            CREATE TABLE IF NOT EXISTS call_logs (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                request_id TEXT NOT NULL,
                rune_name TEXT NOT NULL,
                mode TEXT NOT NULL CHECK(mode IN ('sync', 'stream', 'async')),
                caster_id TEXT,
                latency_ms INTEGER NOT NULL DEFAULT 0,
                status_code INTEGER NOT NULL DEFAULT 200,
                input_size INTEGER NOT NULL DEFAULT 0,
                output_size INTEGER NOT NULL DEFAULT 0,
                timestamp TEXT NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ', 'now'))
            );
            CREATE INDEX IF NOT EXISTS idx_logs_rune ON call_logs(rune_name);
            CREATE INDEX IF NOT EXISTS idx_logs_ts ON call_logs(timestamp);

            CREATE TABLE IF NOT EXISTS rune_snapshots (
                rune_name TEXT PRIMARY KEY,
                version TEXT NOT NULL DEFAULT '',
                description TEXT NOT NULL DEFAULT '',
                supports_stream INTEGER NOT NULL DEFAULT 0,
                gate_path TEXT,
                gate_method TEXT,
                last_seen TEXT NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ', 'now'))
            );"
        )?;
        info!("database migrations complete");
        Ok(())
    }

    /// Access the connection (for sub-modules).
    pub(crate) fn conn(&self) -> std::sync::MutexGuard<Connection> {
        self.conn.lock().unwrap()
    }
}
```

```rust
impl RuneStore {
    /// Create a new API key. Returns (id, raw_key).
    /// Raw key format: rk_{32_hex_chars}
    pub fn create_key(&self, key_type: KeyType, label: &str) -> Result<(i64, String), rusqlite::Error> {
        let raw_bytes: [u8; 16] = rand::rng().random();
        let raw_key = format!("rk_{}", hex::encode(raw_bytes));
        let key_hash = hash_key(&raw_key);
        let key_prefix = &raw_key[..11]; // "rk_" + 8 chars

        let conn = self.conn();
        conn.execute(
            "INSERT INTO api_keys (key_prefix, key_hash, key_type, label) VALUES (?1, ?2, ?3, ?4)",
            params![key_prefix, key_hash, key_type.as_str(), label],
        )?;
        let id = conn.last_insert_rowid();
        Ok((id, raw_key))
    }

    /// Verify a raw key against stored hash. Returns false if not found or revoked.
    pub fn verify_key(&self, raw_key: &str, expected_type: KeyType) -> Result<bool, rusqlite::Error> {
        let key_hash = hash_key(raw_key);
        let conn = self.conn();
        let count: i64 = conn.query_row(
            "SELECT COUNT(*) FROM api_keys WHERE key_hash = ?1 AND key_type = ?2 AND revoked_at IS NULL",
            params![key_hash, expected_type.as_str()],
            |row| row.get(0),
        )?;
        Ok(count > 0)
    }

    /// List all keys (hashes redacted for security).
    pub fn list_keys(&self) -> Result<Vec<ApiKey>, rusqlite::Error> {
        let conn = self.conn();
        let mut stmt = conn.prepare(
            "SELECT id, key_prefix, key_type, label, created_at, revoked_at FROM api_keys ORDER BY id"
        )?;
        let keys = stmt.query_map([], |row| {
            Ok(ApiKey {
                id: row.get(0)?,
                key_prefix: row.get(1)?,
                key_hash: "[redacted]".to_string(),
                key_type: KeyType::from_str(&row.get::<_, String>(2)?).unwrap_or(KeyType::Gate),
                label: row.get(3)?,
                created_at: row.get(4)?,
                revoked_at: row.get(5)?,
            })
        })?.collect::<Result<Vec<_>, _>>()?;
        Ok(keys)
    }

    /// Revoke a key by ID.
    pub fn revoke_key(&self, key_id: i64) -> Result<(), rusqlite::Error> {
        let conn = self.conn();
        conn.execute(
            "UPDATE api_keys SET revoked_at = strftime('%Y-%m-%dT%H:%M:%SZ', 'now') WHERE id = ?1",
            params![key_id],
        )?;
        Ok(())
    }
}

fn hash_key(raw_key: &str) -> String {
    let mut hasher = Sha256::new();
    hasher.update(raw_key.as_bytes());
    hex::encode(hasher.finalize())
}
```

```rust
async fn auth_middleware(
    State(state): State<GateState>,
    req: axum::extract::Request,
    next: axum::middleware::Next,
) -> axum::response::Response {
    if !state.auth_enabled {
        return next.run(req).await;
    }

    let path = req.uri().path().to_string();
    if state.exempt_routes.iter().any(|r| path.starts_with(r)) {
        return next.run(req).await;
    }

    let auth_header = req.headers()
        .get("authorization")
        .and_then(|v| v.to_str().ok())
        .unwrap_or("");

    let key = auth_header.strip_prefix("Bearer ").unwrap_or("");
    if key.is_empty() || !state.key_verifier.verify_gate_key(key).await {
        return axum::response::Response::builder()
            .status(401)
            .header("content-type", "application/json")
            .body(axum::body::Body::from(r#"{"error":"unauthorized","message":"invalid or missing API key"}"#))
            .unwrap();
    }

    next.run(req).await
}
```
