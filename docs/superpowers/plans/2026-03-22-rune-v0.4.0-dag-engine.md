# Rune v0.4.0: DAG Flow Engine Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox syntax for tracking.

**Goal:** Replace the sequential Chain execution in `rune-flow` with a full DAG-based Flow engine — supporting parallel step execution, conditional branching (CEL), input mapping, flow persistence, and Gate API endpoints.

**Architecture:** Seven layers built in dependency order: (1) DAG data structures with `FlowDefinition`/`StepDefinition`; (2) DAG validation with topological sort and cycle detection; (3) DAG execution engine using `tokio::JoinSet` for layer-parallel execution; (4) conditional branching via `cel-interpreter` with skip propagation; (5) `input_mapping` with field-path resolution and `$input` reference; (6) SQLite persistence for flow definitions and execution records; (7) Gate API for CRUD + sync/async/stream execution; (8) async task tracking and SSE streaming.

**Tech Stack:** Rust, tokio (async/parallel), cel-interpreter (CEL expressions), serde_json (data mapping), rusqlite (persistence), axum (API), axum/SSE (streaming)

**Spec:** `docs/superpowers/specs/2026-03-22-rune-v1.0.0-design.md` — Module 5

```rust
/// A condition expression evaluated against upstream outputs.
/// Uses CEL (Common Expression Language) syntax.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Condition {
    /// CEL expression string, e.g. "step_a.output.status == 'ok'"
    pub expression: String,
}

/// Describes how a step's input is assembled from upstream outputs and flow input.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct InputMapping {
    /// Maps target field name -> source expression.
    /// Source expressions:
    ///   - "$input" — the original flow input (entire object)
    ///   - "$input.field.path" — a specific field from the flow input
    ///   - "step_name.output" — entire output of a step
    ///   - "step_name.output.field.path" — a specific field from a step's output
    pub mappings: HashMap<String, String>,
}

/// Definition of a single step within a Flow DAG.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct StepDefinition {
    /// Unique name of this step within the flow.
    pub name: String,
    /// The Rune to invoke for this step.
    pub rune_name: String,
    /// Steps that must complete before this step can run.
    /// Empty = root step (no dependencies).
    #[serde(default)]
    pub depends_on: Vec<String>,
    /// Optional condition — if evaluates to false, step is skipped.
    #[serde(default)]
    pub condition: Option<Condition>,
    /// Optional input mapping — how to assemble this step's input.
    /// If None and single upstream: upstream output is passed directly.
    /// If None and multiple upstreams: registration error.
    #[serde(default)]
    pub input_mapping: Option<InputMapping>,
    /// Step-level timeout override in seconds.
    #[serde(default = "default_step_timeout")]
    pub timeout_secs: u64,
}

fn default_step_timeout() -> u64 {
    30
```

```rust
use crate::dag::{FlowDefinition, StepDefinition, InputMapping};
use std::collections::{HashMap, HashSet, VecDeque};
use thiserror::Error;

#[derive(Debug, Error)]
pub enum ValidationError {
    #[error("flow has no steps")]
    EmptyFlow,
    #[error("duplicate step name: '{0}'")]
    DuplicateStepName(String),
    #[error("step '{step}' depends on unknown step '{dependency}'")]
    UnknownDependency { step: String, dependency: String },
    #[error("cycle detected involving steps: {0:?}")]
    CycleDetected(Vec<String>),
    #[error("step '{step}' has multiple upstreams but no input_mapping")]
    MissingInputMapping { step: String },
}

/// Validates a FlowDefinition and returns topologically sorted layers.
/// Each layer is a Vec of step names that can execute in parallel.
/// Returns ValidationError if the flow is invalid.
pub fn validate_and_sort(flow: &FlowDefinition) -> Result<Vec<Vec<String>>, ValidationError> {
    if flow.steps.is_empty() {
        return Err(ValidationError::EmptyFlow);
    }

    // Check duplicate step names
    let mut seen = HashSet::new();
    for step in &flow.steps {
        if !seen.insert(&step.name) {
            return Err(ValidationError::DuplicateStepName(step.name.clone()));
        }
    }

    // Build adjacency info
    let step_names: HashSet<&str> = flow.steps.iter().map(|s| s.name.as_str()).collect();
    let mut in_degree: HashMap<&str, usize> = HashMap::new();
    let mut dependents: HashMap<&str, Vec<&str>> = HashMap::new();

    for step in &flow.steps {
        in_degree.entry(step.name.as_str()).or_insert(0);
        for dep in &step.depends_on {
            if !step_names.contains(dep.as_str()) {
                return Err(ValidationError::UnknownDependency {
                    step: step.name.clone(),
                    dependency: dep.clone(),
                });
            }
            *in_degree.entry(step.name.as_str()).or_insert(0) += 1;
            dependents.entry(dep.as_str()).or_default().push(step.name.as_str());
        }
    }

    // Check: multiple upstreams without input_mapping
    let step_map: HashMap<&str, &StepDefinition> =
        flow.steps.iter().map(|s| (s.name.as_str(), s)).collect();
    for step in &flow.steps {
        if step.depends_on.len() > 1 && step.input_mapping.is_none() {
            return Err(ValidationError::MissingInputMapping {
                step: step.name.clone(),
            });
        }
    }

    // Kahn's algorithm — topological sort by layers
    let mut queue: VecDeque<&str> = in_degree.iter()
        .filter(|(_, &deg)| deg == 0)
        .map(|(&name, _)| name)
        .collect();

    let mut layers: Vec<Vec<String>> = Vec::new();
    let mut processed = 0usize;

    while !queue.is_empty() {
        let layer: Vec<&str> = queue.drain(..).collect();
        processed += layer.len();

        let mut next_queue = Vec::new();
        for &node in &layer {
            if let Some(deps) = dependents.get(node) {
                for &dep in deps {
                    let deg = in_degree.get_mut(dep).unwrap();
                    *deg -= 1;
                    if *deg == 0 {
                        next_queue.push(dep);
                    }
                }
            }
        }

        layers.push(layer.into_iter().map(|s| s.to_string()).collect());
        queue.extend(next_queue);
    }

    if processed != flow.steps.len() {
        // Find nodes in cycle (those not processed)
        let processed_set: HashSet<&str> = layers.iter()
            .flat_map(|l| l.iter().map(|s| s.as_str()))
            .collect();
        let cycle_nodes: Vec<String> = flow.steps.iter()
            .filter(|s| !processed_set.contains(s.name.as_str()))
            .map(|s| s.name.clone())
            .collect();
        return Err(ValidationError::CycleDetected(cycle_nodes));
    }

    Ok(layers)
}
```

```rust
use std::collections::HashMap;
use std::sync::Arc;
use bytes::Bytes;
use tokio::task::JoinSet;

use rune_core::relay::Relay;
use rune_core::resolver::Resolver;
use rune_core::rune::{RuneContext, RuneError};

use crate::dag::*;
use crate::validate::{validate_and_sort, ValidationError};

#[derive(Debug, thiserror::Error)]
pub enum FlowError {
    #[error("validation failed: {0}")]
    Validation(#[from] ValidationError),

    #[error("flow '{0}' not found")]
    FlowNotFound(String),

    #[error("step '{step_name}' failed: {source}")]
    StepFailed {
        step_name: String,
        source: RuneError,
    },

    #[error("step '{step_name}': rune '{rune_name}' not found")]
    RuneNotFound {
        step_name: String,
        rune_name: String,
    },

    #[error("step '{0}' panicked during execution")]
    StepPanicked(String),
}

/// Stores outputs of completed steps during execution.
pub type StepOutputs = HashMap<String, Bytes>;

/// The DAG Flow execution engine.
pub struct FlowEngine {
    relay: Arc<Relay>,
    resolver: Arc<dyn Resolver>,
}

impl FlowEngine {
    pub fn new(relay: Arc<Relay>, resolver: Arc<dyn Resolver>) -> Self {
        Self { relay, resolver }
    }

    /// Execute a flow definition with the given input.
    /// Returns the execution result including per-step outputs.
    pub async fn execute(
        &self,
        flow: &FlowDefinition,
        input: Bytes,
    ) -> Result<FlowExecutionResult, FlowError> {
        let layers = validate_and_sort(flow)?;
        let execution_id = uuid::Uuid::new_v4().to_string();

        let mut step_outputs: StepOutputs = HashMap::new();
        let mut step_results: Vec<StepResult> = Vec::new();
        let step_map: HashMap<&str, &StepDefinition> =
            flow.steps.iter().map(|s| (s.name.as_str(), s)).collect();

        for layer in &layers {
            let mut join_set: JoinSet<Result<(String, Bytes), FlowError>> = JoinSet::new();

            for step_name in layer {
                let step_def = step_map[step_name.as_str()];

                // Determine step input:
                // - No deps: use flow input
                // - Single dep: use upstream output
                // - Multiple deps: requires input_mapping (handled in Task 5, for now error)
                let step_input = if step_def.depends_on.is_empty() {
                    input.clone()
                } else if step_def.depends_on.len() == 1 {
                    step_outputs
                        .get(&step_def.depends_on[0])
                        .cloned()
                        .unwrap_or_default()
                } else {
                    // Multi-upstream: for now, pass empty (will be replaced by input_mapping)
                    Bytes::from("{}")
                };

                let invoker = self.relay.resolve(&step_def.rune_name, &*self.resolver)
                    .ok_or_else(|| FlowError::RuneNotFound {
                        step_name: step_name.clone(),
                        rune_name: step_def.rune_name.clone(),
                    })?;

                let name = step_name.clone();
                let rune_name = step_def.rune_name.clone();
                let timeout = std::time::Duration::from_secs(step_def.timeout_secs);
                let exec_id = execution_id.clone();

                join_set.spawn(async move {
                    let ctx = RuneContext {
                        rune_name: rune_name.clone(),
                        request_id: format!("f-{}-{}", exec_id, name),
                        context: Default::default(),
                        timeout,
                    };

                    let output = invoker.invoke_once(ctx, step_input).await
                        .map_err(|e| FlowError::StepFailed {
                            step_name: name.clone(),
                            source: e,
                        })?;

                    Ok((name, output))
                });
            }

            // Collect results from this layer
            while let Some(result) = join_set.join_next().await {
                match result {
                    Ok(Ok((name, output))) => {
                        step_results.push(StepResult {
                            name: name.clone(),
                            status: StepStatus::Completed,
                            output: serde_json::from_slice(&output).ok(),
                            error: None,
                        });
                        step_outputs.insert(name, output);
                    }
                    Ok(Err(e)) => {
                        // A step failed — record it and abort
                        let step_name = match &e {
                            FlowError::StepFailed { step_name, .. } => step_name.clone(),
                            FlowError::RuneNotFound { step_name, .. } => step_name.clone(),
                            _ => "unknown".to_string(),
                        };
                        step_results.push(StepResult {
                            name: step_name,
                            status: StepStatus::Failed,
                            output: None,
                            error: Some(e.to_string()),
                        });
                        return Ok(FlowExecutionResult {
                            execution_id,
                            flow_name: flow.name.clone(),
                            success: false,
                            step_results,
                            output: None,
                        });
                    }
                    Err(join_err) => {
                        // Task panicked
                        return Ok(FlowExecutionResult {
                            execution_id,
                            flow_name: flow.name.clone(),
                            success: false,
                            step_results,
                            output: Some(serde_json::json!({
                                "error": format!("step panicked: {}", join_err)
                            })),
                        });
                    }
                }
            }
        }

        // Determine final output: output of the last layer's steps
        // If single terminal step, use its output. If multiple, merge into object.
        let last_layer = layers.last().unwrap();
        let final_output = if last_layer.len() == 1 {
            step_outputs.get(&last_layer[0])
                .and_then(|b| serde_json::from_slice(b).ok())
        } else {
            let mut merged = serde_json::Map::new();
            for name in last_layer {
                if let Some(bytes) = step_outputs.get(name) {
                    if let Ok(val) = serde_json::from_slice::<serde_json::Value>(bytes) {
                        merged.insert(name.clone(), val);
                    }
                }
            }
            Some(serde_json::Value::Object(merged))
        };

        Ok(FlowExecutionResult {
            execution_id,
            flow_name: flow.name.clone(),
            success: true,
            step_results,
            output: final_output,
        })
    }
}
```

```rust
use cel_interpreter::{Context, Program, Value};
use serde_json;

#[derive(Debug, thiserror::Error)]
pub enum CelError {
    #[error("CEL parse error: {0}")]
    ParseError(String),
    #[error("CEL evaluation error: {0}")]
    EvalError(String),
    #[error("CEL result is not a boolean: {0:?}")]
    NotBoolean(String),
}

/// Evaluate a CEL expression against a context of step outputs.
/// The context maps step names to their output JSON values.
/// Returns true if the condition is met, false otherwise.
pub fn evaluate_condition(
    expression: &str,
    step_outputs: &std::collections::HashMap<String, serde_json::Value>,
    flow_input: &serde_json::Value,
) -> Result<bool, CelError> {
    let program = Program::compile(expression)
        .map_err(|e| CelError::ParseError(format!("{:?}", e)))?;

    let mut ctx = Context::default();

    // Add step outputs as variables: each step_name maps to its output JSON
    for (step_name, output) in step_outputs {
        let cel_val = json_to_cel(output);
        ctx.add_variable(step_name, cel_val)
            .map_err(|e| CelError::EvalError(format!("{:?}", e)))?;
    }

    // Add $input as "input" variable
    let input_cel = json_to_cel(flow_input);
    ctx.add_variable("input", input_cel)
        .map_err(|e| CelError::EvalError(format!("{:?}", e)))?;

    let result = program.execute(&ctx)
        .map_err(|e| CelError::EvalError(format!("{:?}", e)))?;

    match result {
        Value::Bool(b) => Ok(b),
        other => Err(CelError::NotBoolean(format!("{:?}", other))),
    }
}

/// Convert serde_json::Value to cel_interpreter::Value
fn json_to_cel(val: &serde_json::Value) -> Value {
    match val {
        serde_json::Value::Null => Value::Null,
        serde_json::Value::Bool(b) => Value::Bool(*b),
        serde_json::Value::Number(n) => {
            if let Some(i) = n.as_i64() {
                Value::Int(i)
            } else if let Some(u) = n.as_u64() {
                Value::UInt(u)
            } else if let Some(f) = n.as_f64() {
                Value::Float(f)
            } else {
                Value::Null
            }
        }
        serde_json::Value::String(s) => Value::String(s.clone().into()),
        serde_json::Value::Array(arr) => {
            Value::List(arr.iter().map(json_to_cel).collect::<Vec<_>>().into())
        }
        serde_json::Value::Object(map) => {
            let cel_map: std::collections::HashMap<String, Value> = map.iter()
                .map(|(k, v)| (k.clone(), json_to_cel(v)))
                .collect();
            Value::Map(cel_map.into_iter().map(|(k, v)| (k.into(), v)).collect())
        }
    }
}
```

- [ ] **Step 2: Add skip propagation logic to execute.rs**

Add skip tracking to the execution engine. The rules are:
1. If a step's condition evaluates to false, mark it as `Skipped`
2. A skipped step's output is treated as `null` for downstream input_mapping
3. If ALL upstreams of a step are skipped, that step is also auto-skipped
4. If some (but not all) upstreams are skipped, the dependency is considered "satisfied"

Update `runtime/rune-flow/src/execute.rs` — add a `skipped: HashSet<String>` to track skipped steps, and before spawning each step, check:
- (a) all deps skipped → auto-skip
- (b) condition evaluates to false → skip

```rust
// In execute(), before the layer loop, add:
let mut skipped: HashSet<String> = HashSet::new();

// In the layer loop, before spawning each step:
// Check auto-skip: all upstream skipped?
let all_deps_skipped = !step_def.depends_on.is_empty()
    && step_def.depends_on.iter().all(|d| skipped.contains(d));

if all_deps_skipped {
    skipped.insert(step_name.clone());
    step_results.push(StepResult {
        name: step_name.clone(),
        status: StepStatus::Skipped,
        output: None,
        error: None,
    });
    continue;
}

// Check condition
if let Some(ref condition) = step_def.condition {
    let cel_outputs: HashMap<String, serde_json::Value> = step_outputs.iter()
        .map(|(k, v)| {
            let json_val = if skipped.contains(k) {
                serde_json::Value::Null
            } else {
                serde_json::from_slice(v).unwrap_or(serde_json::Value::Null)
            };
            (k.clone(), json_val)
        })
        .collect();
    let flow_input_json: serde_json::Value = serde_json::from_slice(&input)
        .unwrap_or(serde_json::Value::Null);

    match crate::cel::evaluate_condition(&condition.expression, &cel_outputs, &flow_input_json) {
        Ok(true) => { /* proceed */ }
        Ok(false) => {
            skipped.insert(step_name.clone());
            step_results.push(StepResult {
                name: step_name.clone(),
                status: StepStatus::Skipped,
                output: None,
                error: None,
            });
            continue;
        }
        Err(e) => {
            step_results.push(StepResult {
                name: step_name.clone(),
                status: StepStatus::Failed,
                output: None,
                error: Some(format!("condition eval error: {}", e)),
            });
            return Ok(FlowExecutionResult {
                execution_id,
                flow_name: flow.name.clone(),
                success: false,
                step_results,
                output: None,
            });
        }
    }
}
```

```rust
use bytes::Bytes;
use serde_json::Value;
use std::collections::HashMap;
use crate::dag::InputMapping;

#[derive(Debug, thiserror::Error)]
pub enum MappingError {
    #[error("invalid source expression: '{0}'")]
    InvalidSource(String),
    #[error("step '{0}' not found in outputs")]
    StepNotFound(String),
    #[error("field path '{path}' not found in step '{step}'")]
    FieldNotFound { step: String, path: String },
    #[error("JSON serialization error: {0}")]
    SerializationError(String),
}

/// Resolve input mapping to produce the input Bytes for a step.
///
/// Source expression syntax:
///   - "$input" — entire flow input
///   - "$input.field.path" — nested field from flow input
///   - "step_name.output" — entire output of a step
///   - "step_name.output.field.path" — nested field from step output
///
/// Skipped steps have null output.
pub fn resolve_mapping(
    mapping: &InputMapping,
    step_outputs: &HashMap<String, Bytes>,
    skipped: &std::collections::HashSet<String>,
    flow_input: &Bytes,
) -> Result<Bytes, MappingError> {
    let flow_input_json: Value = serde_json::from_slice(flow_input)
        .unwrap_or(Value::Null);

    let mut result = serde_json::Map::new();

    for (target_field, source_expr) in &mapping.mappings {
        let value = resolve_source(source_expr, step_outputs, skipped, &flow_input_json)?;
        result.insert(target_field.clone(), value);
    }

    let bytes = serde_json::to_vec(&Value::Object(result))
        .map_err(|e| MappingError::SerializationError(e.to_string()))?;
    Ok(Bytes::from(bytes))
}

/// Resolve a single source expression to a JSON value.
fn resolve_source(
    expr: &str,
    step_outputs: &HashMap<String, Bytes>,
    skipped: &std::collections::HashSet<String>,
    flow_input: &Value,
) -> Result<Value, MappingError> {
    if expr == "$input" {
        return Ok(flow_input.clone());
    }

    if let Some(path) = expr.strip_prefix("$input.") {
        return Ok(resolve_field_path(flow_input, path));
    }

    // Parse "step_name.output" or "step_name.output.field.path"
    let parts: Vec<&str> = expr.splitn(3, '.').collect();
    if parts.len() < 2 || parts[1] != "output" {
        return Err(MappingError::InvalidSource(expr.to_string()));
    }

    let step_name = parts[0];

    // Skipped step → null
    if skipped.contains(step_name) {
        return Ok(Value::Null);
    }

    let step_output = step_outputs.get(step_name)
        .ok_or_else(|| MappingError::StepNotFound(step_name.to_string()))?;

    let step_json: Value = serde_json::from_slice(step_output)
        .unwrap_or(Value::Null);

    if parts.len() == 2 {
        // "step_name.output" — return entire output
        Ok(step_json)
    } else {
        // "step_name.output.field.path" — resolve nested path
        Ok(resolve_field_path(&step_json, parts[2]))
    }
}

/// Resolve a dot-separated field path against a JSON value.
/// Returns null if path doesn't exist.
fn resolve_field_path(value: &Value, path: &str) -> Value {
    let mut current = value;
    for segment in path.split('.') {
        match current {
            Value::Object(map) => {
                current = map.get(segment).unwrap_or(&Value::Null);
            }
            Value::Array(arr) => {
                if let Ok(idx) = segment.parse::<usize>() {
                    current = arr.get(idx).unwrap_or(&Value::Null);
                } else {
                    return Value::Null;
                }
            }
            _ => return Value::Null,
        }
    }
    current.clone()
}
```

```rust
             VALUES (?1, ?2, ?3, ?4)
             ON CONFLICT(name) DO UPDATE SET definition = ?2, updated_at = ?4",

        )?;
        Ok(())

    pub fn get_flow_definition(&self, name: &str) -> StoreResult<Option<String>> {
        let conn = self.conn.lock().unwrap();
        let mut stmt = conn.prepare(
            "SELECT definition FROM flow_definitions WHERE name = ?1",
        )?;
        let result = stmt.query_row(rusqlite::params![name], |row| {
            row.get::<_, String>(0)
        });
        match result {
            Ok(def) => Ok(Some(def)),
            Err(rusqlite::Error::QueryReturnedNoRows) => Ok(None),
            Err(e) => Err(e.into()),
        }

                row.get::<_, String>(1)?,
```

```rust
async fn register_flow(
    State(state): State<GateState>,
    Json(body): Json<serde_json::Value>,
) -> impl IntoResponse {
    // Parse and validate the flow definition
    let flow_def: rune_flow::dag::FlowDefinition = match serde_json::from_value(body.clone()) {
        Ok(f) => f,
        Err(e) => {
            return error_response(StatusCode::BAD_REQUEST, "INVALID_FLOW", &e.to_string());
        }
    };

    // Validate DAG structure
    if let Err(e) = rune_flow::validate::validate_and_sort(&flow_def) {
        return error_response(StatusCode::BAD_REQUEST, "INVALID_DAG", &e.to_string());
    }

    // Persist
    let json_str = serde_json::to_string(&flow_def).unwrap();
    if let Err(e) = state.store.save_flow_definition(&flow_def.name, &json_str) {
        return error_response(
            StatusCode::INTERNAL_SERVER_ERROR,
            "STORE_ERROR",
            &e.to_string(),
        );
    }

    (StatusCode::CREATED, Json(serde_json::json!({
        "name": flow_def.name,
        "steps": flow_def.steps.len(),
        "status": "registered"
    }))).into_response()
}

async fn list_flows(State(state): State<GateState>) -> impl IntoResponse {
    match state.store.list_flow_definitions() {
        Ok(flows) => {
            let items: Vec<serde_json::Value> = flows.into_iter()
                .map(|(name, created, updated)| serde_json::json!({
                    "name": name,
                    "created_at": created,
                    "updated_at": updated,
                }))
                .collect();
            Json(serde_json::json!({ "flows": items })).into_response()
        }
        Err(e) => error_response(
            StatusCode::INTERNAL_SERVER_ERROR,
            "STORE_ERROR",
            &e.to_string(),
        ),
    }
}

async fn get_flow(
    State(state): State<GateState>,
    Path(name): Path<String>,

        Ok(Some(json_str)) => {

            Json(val).into_response()
        }
        Ok(None) => error_response(StatusCode::NOT_FOUND, "NOT_FOUND", &format!("flow '{}' not found", name)),
        Err(e) => error_response(StatusCode::INTERNAL_SERVER_ERROR, "STORE_ERROR", &e.to_string()),
    }

        Ok(false) => error_response(StatusCode::NOT_FOUND, "NOT_FOUND", &format!("flow '{}' not found", name)),
        Err(e) => error_response(StatusCode::INTERNAL_SERVER_ERROR, "STORE_ERROR", &e.to_string()),
    }
}

async fn run_flow(
    State(state): State<GateState>,
    Path(name): Path<String>,
    Query(params): Query<RunParams>,
    body: Bytes,
) -> impl IntoResponse {
    // Load flow definition
    let flow_json = match state.store.get_flow_definition(&name) {
        Ok(Some(j)) => j,
        Ok(None) => {
            return error_response(StatusCode::NOT_FOUND, "NOT_FOUND", &format!("flow '{}' not found", name));
        }
        Err(e) => {
            return error_response(StatusCode::INTERNAL_SERVER_ERROR, "STORE_ERROR", &e.to_string());
        }
    };

    let flow_def: rune_flow::dag::FlowDefinition = match serde_json::from_str(&flow_json) {
        Ok(f) => f,
        Err(e) => {
            return error_response(StatusCode::INTERNAL_SERVER_ERROR, "PARSE_ERROR", &e.to_string());
        }
    };

    let input = if body.is_empty() { Bytes::from("{}") } else { body };

    // Sync execution (async and stream modes handled in Task 8)
    match state.flow_engine.execute(&flow_def, input).await {
        Ok(result) => {
            // Persist execution record
            let _ = state.store.insert_flow_execution(
                &result.execution_id, &name,
                Some(&serde_json::to_string(&result.output).unwrap_or_default()),
            );
            let status = if result.success {
                rune_store::FlowExecutionStatus::Completed
            } else {
                rune_store::FlowExecutionStatus::Failed
            };
            let _ = state.store.update_flow_execution(
                &result.execution_id, status,
                result.output.as_ref().map(|v| serde_json::to_string(v).unwrap_or_default()).as_deref(),
                None,
                Some(&serde_json::to_string(&result.step_results).unwrap_or_default()),
            );

            let status_code = if result.success { StatusCode::OK } else { StatusCode::INTERNAL_SERVER_ERROR };
            (status_code, Json(serde_json::json!({
                "execution_id": result.execution_id,
                "flow_name": result.flow_name,
                "success": result.success,
                "output": result.output,
                "step_results": result.step_results,
            }))).into_response()
        }
        Err(e) => {
            error_response(StatusCode::INTERNAL_SERVER_ERROR, "FLOW_ERROR", &e.to_string())
        }
    }
}
```

```rust
            }
        }
```
```rust
    };
```
